Get Started

  1. Add Markup

    <!-- Set the video url to the data-src property -->
<video>
	<source data-src="vdo.mp4" type="video/mp4">
</video>

  2. Add JS

    <!-- Don't add my CDN in your project. I recommend adding it to your CDN -->
<script src="https://cdn.jsdelivr.net/gh/fe-jw/vdOptimizer/vdOptimizer.min.js"></script>

  3. Initialize

    //You have to call funtion after the window load
window.addEventListener('load', function(){
	vdOptimizer();
});
